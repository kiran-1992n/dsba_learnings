List Copy Operation:

	l1 = [1,2,3,4]
	l2 = l1
	print(l1) # [1,2,3,4]
	print(l2) # [1,2,3,4]
	l2[0] = 100
	print(l1) # [100,2,3,4]
	print(l2) # [100,2,3,4]

	-----------------------

	l1 = [1,2,3,4]
	l2 = l1.copy()
	print(l1) # [1,2,3,4]
	print(l2) # [1,2,3,4]
	l2[0] = 100
	print(l1) # [1,2,3,4]
	print(l2) # [100,2,3,4]
